% 18.5.9: Hermite
\[ H_n(x) = (-1)^n e^{x^2} \frac{d^n}{dx^n} e^{-x^2} \]

% 18.5.12: Laguerre
\[ L_n(x) = \frac{e^x}{n!} \frac{d^n}{dx^n}(x^n e^{-x}) \]

% 18.3.1: Chebyshev T
\[ T_n(x) = \cos(n\arccos x) \]

% 18.3.2: Chebyshev U
\[ U_n(x) = \frac{\sin((n+1)\arccos x)}{\sin(\arccos x)} \]
