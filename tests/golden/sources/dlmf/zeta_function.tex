% 25.2.1: Zeta series
\[ \zeta(s) = \sum_{n=1}^\infty \frac{1}{n^s} \]

% 25.5.1: Integral
\[ \zeta(s) = \frac{1}{\Gamma(s)} \int_0^\infty \frac{t^{s-1}}{e^t-1} \, dt \]

% 25.4.1: Functional eq (simplified)
\[ \zeta(s)\zeta(s-1) = \frac{\pi^s}{\Gamma(1-s)} \]

% 25.6.1: Basel problem
\[ \zeta(2) = \frac{\pi^2}{6} \]

% 25.7.1: Dirichlet eta
\[ \eta(s) = \sum_{n=1}^\infty \frac{(-1)^{n+1}}{n^s} = (1-2^{1-s})\zeta(s) \]
