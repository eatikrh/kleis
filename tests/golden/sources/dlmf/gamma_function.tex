% 5.2.1: Gamma integral
\[ \Gamma(z) = \int_0^\infty t^{z-1} e^{-t} \, dt \]

% 5.5.1: Recurrence
\[ \Gamma(z+1) = z\Gamma(z) \]

% 5.5.3: Reflection
\[ \Gamma(z)\Gamma(1-z) = \frac{\pi}{\sin(\pi z)} \]

% 5.5.5: Half-integer
\[ \Gamma\left(\frac{1}{2}\right) = \sqrt{\pi} \]

% 5.2.2: Digamma
\[ \psi(z) = \frac{d}{dz}\ln\Gamma(z) = \frac{\Gamma'(z)}{\Gamma(z)} \]
