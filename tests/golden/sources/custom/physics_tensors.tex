% Physics and Tensor Notation Examples
% Standard notation from GR and QFT textbooks

% === GENERAL RELATIVITY ===

% Einstein Field Equations (full form)
G_{\mu\nu} + \Lambda g_{\mu\nu} = \frac{8\pi G}{c^4} T_{\mu\nu}

% Christoffel symbols (Levi-Civita connection)
\Gamma^\rho_{\mu\nu} = \frac{1}{2} g^{\rho\sigma} \left( \partial_\mu g_{\nu\sigma} + \partial_\nu g_{\mu\sigma} - \partial_\sigma g_{\mu\nu} \right)

% Riemann curvature tensor
R^\rho_{\sigma\mu\nu} = \partial_\mu \Gamma^\rho_{\nu\sigma} - \partial_\nu \Gamma^\rho_{\mu\sigma} + \Gamma^\rho_{\mu\lambda} \Gamma^\lambda_{\nu\sigma} - \Gamma^\rho_{\nu\lambda} \Gamma^\lambda_{\mu\sigma}

% Ricci tensor
R_{\mu\nu} = R^\rho_{\mu\rho\nu}

% Ricci scalar
R = g^{\mu\nu} R_{\mu\nu}

% Covariant derivative
\nabla_\mu V^\nu

% === ELECTROMAGNETISM ===

% Maxwell equations (tensor form)
\partial_\mu F^{\mu\nu} = \mu_0 J^\nu

% Field tensor from potential
F_{\mu\nu} = \partial_\mu A_\nu - \partial_\nu A_\mu

% Electromagnetic stress-energy tensor
T^{\mu\nu} = \frac{1}{\mu_0} \left( F^{\mu\alpha} F^\nu_{\ \alpha} - \frac{1}{4} g^{\mu\nu} F^{\alpha\beta} F_{\alpha\beta} \right)

% === QUANTUM FIELD THEORY ===

% Klein-Gordon equation
\left( \Box + \frac{m^2 c^2}{\hbar^2} \right) \phi = 0

% Dirac equation
\left( i\hbar \gamma^\mu \partial_\mu - mc \right) \psi = 0

% Commutator
[A, B] = AB - BA

% Anticommutator
\{A, B\} = AB + BA

% === KALUZA-KLEIN THEORY ===

% 5D metric ansatz
\hat{g}_{AB} = \begin{pmatrix}
g_{\mu\nu} + \phi^2 A_\mu A_\nu & \phi^2 A_\mu \\
\phi^2 A_\nu & \phi^2
\end{pmatrix}

% === INDEX NOTATION ===

% Mixed indices
T^\mu_{\ \nu}

% Multiple indices
R^{\mu\nu\rho\sigma}

% Contraction
T^\mu_{\ \mu}
